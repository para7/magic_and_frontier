package views

templ Layout(title string, currentPath string) {
	<!doctype html>
	<html lang="ja">
		<head>
			<meta charset="UTF-8"/>
			<meta name="viewport" content="width=device-width, initial-scale=1.0"/>
			<title>{ title }</title>
			<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/@picocss/pico@2/css/pico.min.css"/>
			<script src="https://unpkg.com/htmx.org@2.0.4"></script>
			<style>
				:root { --sidebar-width: 16rem; }
				.layout-shell { display: grid; grid-template-columns: var(--sidebar-width) 1fr; min-height: 100vh; }
				.sidebar { border-right: 1px solid var(--pico-muted-border-color); padding: 1.5rem 1rem; }
				.sidebar h3 { margin-bottom: 0.3rem; }
				.sidebar p { margin-bottom: 1rem; color: var(--pico-muted-color); font-size: 0.92rem; }
				.sidebar nav { margin-bottom: 1rem; }
				.sidebar a { display: block; padding: 0.5rem 0.75rem; border-radius: 8px; text-decoration: none; }
				.sidebar a.active { background: var(--pico-primary-background); color: var(--pico-primary-inverse); }
				.content { padding: 2rem 1.5rem; }
				#toast { position: fixed; right: 1rem; top: 1rem; min-width: 260px; z-index: 1000; }
				.toast-item { padding: 0.75rem 1rem; border-radius: 8px; margin-bottom: 0.5rem; background: var(--pico-primary-background); color: var(--pico-primary-inverse); }
				@media (max-width: 900px) {
					.layout-shell { grid-template-columns: 1fr; }
					.sidebar { border-right: 0; border-bottom: 1px solid var(--pico-muted-border-color); }
				}
			</style>
		</head>
		<body>
			<div class="layout-shell">
				<aside class="sidebar">
					<h3>Tools2</h3>
					<p>設定を入力して、そのまま保存できます。</p>
					<nav>
						if currentPath == "/form" {
							<a href="/form" class="active">フォーム</a>
						} else {
							<a href="/form">フォーム</a>
						}
					</nav>
				</aside>
				<main class="content">
					{ children... }
				</main>
			</div>
			<div id="toast"></div>
			<script>
				document.body.addEventListener('htmx:afterRequest', function (evt) {
					if (!evt.detail || !evt.detail.xhr) {
						return;
					}
					var trigger = evt.detail.xhr.getResponseHeader('HX-Trigger');
					if (!trigger) {
						return;
					}
					var toast = document.getElementById('toast');
					if (!toast) {
						return;
					}
					var item = document.createElement('div');
					item.className = 'toast-item';
					item.textContent = trigger;
					toast.appendChild(item);
					window.setTimeout(function () {
						item.remove();
					}, 3000);
				});
			</script>
		</body>
	</html>
}
